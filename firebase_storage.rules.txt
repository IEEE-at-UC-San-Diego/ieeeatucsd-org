rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Helper function to check if user can manage events (officers or active event submitters)
    function canManageEventFiles() {
      return request.auth != null &&
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in [
          'General Officer', 'Executive Officer', 'Administrator'
        ];
    }

    // Match any file path that starts with the user's UID
    match /resumes/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Invoice files - users can upload their own, officers and event submitters can read/write all
    match /invoices/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == userId || canManageEventFiles());
    }

    // Room booking files - users can upload their own, officers and event submitters can read/write all
    match /room_bookings/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == userId || canManageEventFiles());
    }

    // Logo files - users can upload their own, officers and event submitters can read/write all
    match /logos/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == userId || canManageEventFiles());
    }
    
    // Reimbursement files - users can upload their own, officers and event submitters can read/write all
    match /reimbursements/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == userId || canManageEventFiles());
    }

    // Fund deposit receipt files - users can upload their own, officers and event submitters can read/write all
    match /fund_deposits/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null &&
        (request.auth.uid == userId || canManageEventFiles());
    }
    
    // Graphics files - officers can upload and read all graphics files
    match /graphics/{userId}/{allPaths=**} {
      allow write: if request.auth != null &&
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in [
          'General Officer', 'Executive Officer', 'Administrator'
        ];
      allow read: if request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in [
          'General Officer', 'Executive Officer', 'Administrator'
        ];
    }
    
    // Default deny all other paths
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
